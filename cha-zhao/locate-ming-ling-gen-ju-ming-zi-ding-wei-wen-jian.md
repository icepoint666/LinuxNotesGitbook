# locate命令——根据名字定位文件

定位

```bash
$ locate tmp
# 会输出所有绝对路径文件名中包含tmp字段的文件或者目录
```

示例：

```bash
$ locate bin/python
/usr/bin/python
/usr/bin/python-config
/usr/bin/python2
/usr/bin/python2-config
/usr/bin/python2.7
/usr/bin/python2.7-config
/usr/bin/python3
/usr/bin/python3-config
/usr/bin/python3-gflags2man
/usr/bin/python3.6
/usr/bin/python3.6-config
/usr/bin/python3.6m
/usr/bin/python3.6m-config
/usr/bin/python3m
/usr/bin/python3m-config
```
